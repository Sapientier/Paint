Uso de teclas:

key = L: Dibujar Linea (Mantener presionado click izquierdo)
key = Q: Dibujar Cuadrado (Mantener presionado click izquierdo)
key = C: Dibujar Circulo (Mantener presionado click izquierdo)
key = E: Dibujar Elipse (Mantener presionado click izquierdo)
key = T: Dibujar Triangulo (Selecionar 3 puntos)
key = B: Dibujar Curva Bezier (Seleccionar N puntos)
key = P: Para empezar a seleccionar figuras y mostrar Bounding Box
key = S: Guarda la escena de dibujo actual en un .json
key = D: Carga la escena desde un .json
key = Z: Mueve hacia adelante las figurar ocultas
key = Del/Supr: Eliminar figura seleccionada
